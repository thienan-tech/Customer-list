CREATE TABLE khachhang (  
  id INT AUTO_INCREMENT PRIMARY KEY,  
  ma_kh VARCHAR(50) UNIQUE NOT NULL,  
  ten_kh VARCHAR(100) NOT NULL,  
  sdt VARCHAR(15) NOT NULL,  
  dia_chi VARCHAR(255) NOT NULL,  
  email VARCHAR(100) NOT NULL,  
  ngaysinh DATE NOT NULL  
);  

CREATE TABLE banggia (  
  id INT AUTO_INCREMENT PRIMARY KEY,  
  ma_vtu VARCHAR(50) UNIQUE NOT NULL,  
  ten_vtu VARCHAR(100) NOT NULL,  
  dvt VARCHAR(20) NOT NULL,  
  don_gia DECIMAL(15, 2) NOT NULL,  
  kl_ton INT NOT NULL  
);  

CREATE TABLE hoadon (  
  id INT AUTO_INCREMENT PRIMARY KEY,  
  ma_kh VARCHAR(50) NOT NULL,  
  tong_dh DECIMAL(15, 2) NOT NULL,  
  cong_no DECIMAL(15, 2) NOT NULL,  
  ma_hd VARCHAR(50) NOT NULL,  
  FOREIGN KEY (ma_kh) REFERENCES khachhang(ma_kh)  
);  

CREATE TABLE ct_hoadon (  
  id INT AUTO_INCREMENT PRIMARY KEY,  
  ma_dh INT NOT NULL,  
  ma_vtu VARCHAR(50) NOT NULL,  
  ten_vtu VARCHAR(100) NOT NULL,  
  dvt VARCHAR(20) NOT NULL,  
  so_luong INT NOT NULL,  
  don_gia DECIMAL(15, 2) NOT NULL,  
  thanh_tien DECIMAL(15, 2) NOT NULL,  
  ghi_chu VARCHAR(255),  
  FOREIGN KEY (ma_vtu) REFERENCES banggia(ma_vtu),  
  FOREIGN KEY (ma_dh) REFERENCES hoadon(id)  
);
